# clickban Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a lazygit-style TUI kanban board that integrates with ClickUp for a specific space, showing leaf tasks as cards with full CRUD capabilities.

**Architecture:** Bubble Tea (Elm architecture) with a root app model that switches between three views (kanban, my-tasks, detail). An API client layer wraps all ClickUp HTTP calls. UI components are pure rendering functions that take data and return styled strings via Lipgloss.

**Tech Stack:** Go, Bubble Tea, Lipgloss, Bubbles, ClickUp API v2

```go
package api

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
)

const defaultBaseURL = "https://api.clickup.com/api/v2"

type Client struct {
	token   string
	baseURL string
	http    *http.Client
}

type Option func(*Client)

func WithBaseURL(url string) Option {
	return func(c *Client) {
		c.baseURL = url
	}
}

func NewClient(token string, opts ...Option) *Client {
	c := &Client{
		token:   token,
		baseURL: defaultBaseURL,
		http:    &http.Client{},
	}
	for _, opt := range opts {
		opt(c)
	}
	return c
}

func (c *Client) Get(path string, result interface{}) error {
	return c.do(http.MethodGet, path, nil, result)
}

func (c *Client) Post(path string, body interface{}, result interface{}) error {
	return c.do(http.MethodPost, path, body, result)
}

func (c *Client) Put(path string, body interface{}, result interface{}) error {
	return c.do(http.MethodPut, path, body, result)
}

func (c *Client) do(method, path string, body interface{}, result interface{}) error {
	var reqBody io.Reader
	if body != nil {
		data, err := json.Marshal(body)
		if err != nil {
			return fmt.Errorf("marshal request body: %w", err)
		}
		reqBody = bytes.NewReader(data)
	}

	req, err := http.NewRequest(method, c.baseURL+path, reqBody)
	if err != nil {
		return fmt.Errorf("create request: %w", err)
	}

	req.Header.Set("Authorization", c.token)
	req.Header.Set("Content-Type", "application/json")

	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("execute request: %w", err)
	}
	defer resp.Body.Close()

	respBody, err := io.ReadAll(resp.Body)
	if err != nil {
		return fmt.Errorf("read response: %w", err)
	}

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		return fmt.Errorf("API error (HTTP %d): %s", resp.StatusCode, string(respBody))
	}

	if result != nil && len(respBody) > 0 {
		if err := json.Unmarshal(respBody, result); err != nil {
			return fmt.Errorf("decode response: %w", err)
		}
	}

	return nil
}
```

```go
func (c *Client) GetTasks(listID string) ([]Task, error) {

	if err := c.Get(fmt.Sprintf("/list/%s/task?subtasks=true", listID), &resp); err != nil {
		return nil, fmt.Errorf("get tasks for list %s: %w", listID, err)
	}
	return resp.Tasks, nil
}

func (c *Client) GetTaskWithSubtasks(taskID string) (*Task, error) {
	var task Task
	if err := c.Get(fmt.Sprintf("/task/%s?include_subtasks=true", taskID), &task); err != nil {
		return nil, fmt.Errorf("get task %s with subtasks: %w", taskID, err)
	}
	return &task, nil
}

func (c *Client) UpdateTask(taskID string, req *UpdateTaskRequest) error {
	var result Task
	if err := c.Put(fmt.Sprintf("/task/%s", taskID), req, &result); err != nil {
		return fmt.Errorf("update task %s: %w", taskID, err)
	}
	return nil
}

// ResolveLeafTasks walks the task tree and returns only leaf tasks (those with no children).
// It uses already-fetched subtask data. If depth limit is reached, the task at that depth
// is treated as a leaf regardless of whether it has children.
func (c *Client) ResolveLeafTasks(tasks []Task, maxDepth int) []Task {
	var leaves []Task
	for _, task := range tasks {
		leaves = append(leaves, resolveLeaves(task, 0, maxDepth)...)
	}
	return leaves
}

func resolveLeaves(task Task, depth, maxDepth int) []Task {
	if len(task.Subtasks) == 0 || depth >= maxDepth {
		return []Task{task}
	}
	var leaves []Task
	for _, sub := range task.Subtasks {
		leaves = append(leaves, resolveLeaves(sub, depth+1, maxDepth)...)
	}
	return leaves
}
```

```go
package ui

import (
	"fmt"
	"strings"

	"github.com/charmbracelet/lipgloss"
	"github.com/nhinkley/clickban/internal/api"
)

// FormatDuration converts milliseconds to a human-readable duration string.
func FormatDuration(ms int64) string {
	if ms <= 0 {
		return ""
	}
	totalMinutes := ms / 60000
	hours := totalMinutes / 60
	minutes := totalMinutes % 60
	if hours > 0 && minutes > 0 {
		return fmt.Sprintf("%dh%dm", hours, minutes)
	}
	if hours > 0 {
		return fmt.Sprintf("%dh", hours)
	}
	return fmt.Sprintf("%dm", minutes)
}

// RenderCard renders a kanban card for a task.
func RenderCard(task api.Task, width int, selected bool) string {
	// Priority color for left border
	priorityNum := 4 // default low
	if task.Priority != nil {
		switch task.Priority.Priority {
		case "urgent":
			priorityNum = 1
		case "high":
			priorityNum = 2
		case "normal":
			priorityNum = 3
		case "low":
			priorityNum = 4
		}
	}
	pColor := PriorityColor(priorityNum)

	// Assignee names
	var assignees []string
	for _, a := range task.Assignees {
		assignees = append(assignees, a.Username)
	}
	assigneeStr := strings.Join(assignees, ", ")

	// Time spent
	timeStr := FormatDuration(task.TimeSpent)

	// Build info line
	var infoParts []string
	if assigneeStr != "" {
		infoParts = append(infoParts, assigneeStr)
	}
	if timeStr != "" {
		infoParts = append(infoParts, timeStr)
	}
	infoLine := strings.Join(infoParts, " · ")

	// Truncate name if needed
	name := task.Name
	maxName := width - 4
	if maxName > 0 && len(name) > maxName {
		name = name[:maxName-1] + "…"
	}

	// Build card content
	content := name
	if infoLine != "" {
		content += "\n" + lipgloss.NewStyle().Foreground(ColorFgDim).Render(infoLine)
	}

	// Card style
	bg := ColorCardBg
	if !selected && task.Priority == nil {
		bg = ColorCardDim
	}

	style := lipgloss.NewStyle().
		Width(width - 2).
		Padding(0, 1).
		Background(bg).
		Foreground(ColorFgBright).
		BorderStyle(lipgloss.Border{Left: "▎"}).
		BorderLeft(true).
		BorderForeground(pColor)

	if selected {
		style = style.
			BorderStyle(lipgloss.Border{
				Left:  "▎",
				Right: "│",
				Top:   "─",
				Bottom: "─",
				TopLeft: "╭",
				TopRight: "╮",
				BottomLeft: "╰",
				BottomRight: "╯",
			}).
			BorderRight(true).
			BorderTop(true).
			BorderBottom(true).
			BorderForeground(ColorBorderAct)
	}

	return style.Render(content)
}
```

```go
// RenderPreview renders the right-side preview pane for a selected task.
func RenderPreview(task api.Task, width, height int) string {
	// Title
	title := TitleStyle.Render(task.Name)

	// Badges
	var badges []string
	if task.Priority != nil {
		badges = append(badges, lipgloss.NewStyle().
			Foreground(ColorRed).
			Render(task.Priority.Priority))
	}
	badges = append(badges, lipgloss.NewStyle().
		Foreground(ColorYellow).
		Render(task.Status.Status))
	if task.TaskType != nil {
		badges = append(badges, lipgloss.NewStyle().
			Foreground(ColorPurple).
			Render(task.TaskType.Name))
	}
	badgeLine := strings.Join(badges, " · ")

	// Assignees and time
	var metaParts []string
	var names []string
	for _, a := range task.Assignees {
		names = append(names, "@"+a.Username)
	}
	if len(names) > 0 {
		metaParts = append(metaParts, strings.Join(names, ", "))
	}
	if task.TimeSpent > 0 {
		metaParts = append(metaParts, FormatDuration(task.TimeSpent))
	}
	metaLine := DimStyle.Render(strings.Join(metaParts, " · "))

	// Description
	desc := task.Description
	maxDescLines := height - 8
	if maxDescLines < 3 {
		maxDescLines = 3
	}
	descLines := strings.Split(desc, "\n")
	if len(descLines) > maxDescLines {
		descLines = descLines[:maxDescLines]
		descLines = append(descLines, DimStyle.Render("..."))
	}
	descStr := strings.Join(descLines, "\n")

	// Assemble
	content := fmt.Sprintf("%s\n%s\n%s\n\n%s\n%s",
		title,
		badgeLine,
		metaLine,
		DimStyle.Render("DESCRIPTION"),
		descStr,
	)

	style := BorderStyle.
		Width(width - 2).
		Height(height - 2).
		Padding(0, 1)

	return style.Render(content)
}
```

```go
package ui

import (
	"fmt"
	"strings"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/charmbracelet/lipgloss"
)

// PickerItem represents a selectable item in the picker.
type PickerItem struct {
	ID    string
	Label string
}

// PickerResult is the message returned when the picker is closed.
type PickerResult struct {
	Selected  []PickerItem
	Cancelled bool
}

// Picker is a dropdown/multi-select component.
type Picker struct {
	title     string
	items     []PickerItem
	cursor    int
	selected  map[int]bool
	multi     bool
}

// NewPicker creates a new picker. If multi is true, allows multiple selection.
func NewPicker(title string, items []PickerItem, multi bool) Picker {
	return Picker{
		title:    title,
		items:    items,
		cursor:   0,
		selected: make(map[int]bool),
		multi:    multi,
	}
}

func (p Picker) CursorIndex() int { return p.cursor }

func (p Picker) IsSelected(index int) bool { return p.selected[index] }

func (p Picker) Init() tea.Cmd { return nil }

func (p Picker) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "j", "down":
			if p.cursor < len(p.items)-1 {
				p.cursor++
			}
		case "k", "up":
			if p.cursor > 0 {
				p.cursor--
			}
		case " ":
			if p.multi {
				p.selected[p.cursor] = !p.selected[p.cursor]
				if !p.selected[p.cursor] {
					delete(p.selected, p.cursor)
				}
			}
		case "enter":
			var result []PickerItem
			if p.multi {
				for i, item := range p.items {
					if p.selected[i] {
						result = append(result, item)
					}
				}
			} else {
				if p.cursor < len(p.items) {
					result = []PickerItem{p.items[p.cursor]}
				}
			}
			return p, func() tea.Msg {
				return PickerResult{Selected: result}
			}
		case "esc":
			return p, func() tea.Msg {
				return PickerResult{Cancelled: true}
			}
		}
	}
	return p, nil
}

func (p Picker) View() string {
	var b strings.Builder

	titleStyle := lipgloss.NewStyle().Foreground(ColorBlue).Bold(true)
	b.WriteString(titleStyle.Render(p.title) + "\n\n")

	for i, item := range p.items {
		cursor := "  "
		if i == p.cursor {
			cursor = lipgloss.NewStyle().Foreground(ColorBlue).Render("> ")
		}

		check := ""
		if p.multi {
			if p.selected[i] {
				check = lipgloss.NewStyle().Foreground(ColorGreen).Render("[x] ")
			} else {
				check = DimStyle.Render("[ ] ")
			}
		}

		label := item.Label
		if i == p.cursor {
			label = lipgloss.NewStyle().Foreground(ColorFgBright).Render(label)
		} else {
			label = DimStyle.Render(label)
		}

		b.WriteString(fmt.Sprintf("%s%s%s\n", cursor, check, label))
	}

	return b.String()
}
```

```go
// EditorResult is the message returned when editing is complete.
type EditorResult struct {
	Value     string
	Cancelled bool
}

// ExternalEditorResult is returned after $EDITOR completes.
type ExternalEditorResult struct {
	Content string
	Err     error
}

// Editor is an inline text editor component.
type Editor struct {
	title string
	input textinput.Model
}

// NewEditor creates a new inline editor with initial content.
func NewEditor(title, initial string) Editor {
	ti := textinput.New()
	ti.SetValue(initial)
	ti.Focus()
	ti.CharLimit = 500
	ti.Width = 60

	return Editor{
		title: title,
		input: ti,
	}
}

func (e Editor) Value() string { return e.input.Value() }

func (e Editor) Init() tea.Cmd { return textinput.Blink }

func (e Editor) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.Type {
		case tea.KeyEnter:
			return e, func() tea.Msg {
				return EditorResult{Value: e.input.Value()}
			}
		case tea.KeyEsc:
			return e, func() tea.Msg {
				return EditorResult{Cancelled: true}
			}
		}
	}

	var cmd tea.Cmd
	e.input, cmd = e.input.Update(msg)
	return e, cmd
}

func (e Editor) View() string {
	titleStyle := lipgloss.NewStyle().Foreground(ColorBlue).Bold(true)
	return fmt.Sprintf("%s\n\n%s\n\n%s",
		titleStyle.Render(e.title),
		e.input.View(),
		DimStyle.Render("enter confirm · esc cancel"),
	)
}

// OpenExternalEditor opens $EDITOR with the given content and returns the result.
// This returns a tea.Cmd that should be executed by the Bubble Tea runtime.
func OpenExternalEditor(content string) tea.Cmd {
	return func() tea.Msg {
		editor := os.Getenv("EDITOR")
		if editor == "" {
			editor = "vi"
		}

		tmpFile, err := os.CreateTemp("", "clickban-*.md")
		if err != nil {
			return ExternalEditorResult{Err: fmt.Errorf("create temp file: %w", err)}
		}
		tmpPath := tmpFile.Name()
		defer os.Remove(tmpPath)

		if _, err := tmpFile.WriteString(content); err != nil {
			tmpFile.Close()
			return ExternalEditorResult{Err: fmt.Errorf("write temp file: %w", err)}
		}
		tmpFile.Close()

		cmd := exec.Command(editor, tmpPath)
		cmd.Stdin = os.Stdin
		cmd.Stdout = os.Stdout
		cmd.Stderr = os.Stderr

		if err := cmd.Run(); err != nil {
			return ExternalEditorResult{Err: fmt.Errorf("editor exited with error: %w", err)}
		}

		result, err := os.ReadFile(tmpPath)
		if err != nil {
			return ExternalEditorResult{Err: fmt.Errorf("read temp file: %w", err)}
		}

		return ExternalEditorResult{Content: string(result)}
	}
}
```

```go
	"github.com/charmbracelet/lipgloss"

// TimerMode represents the current time entry mode.
type TimerMode int

const (
	TimerModeMenu TimerMode = iota
	TimerModeLive
	TimerModeDuration
	TimerModeRange
)

// TimerResult is returned when time entry is complete.
type TimerResult struct {
	Mode      TimerMode
	Duration  int64 // milliseconds
	Start     int64 // unix ms (for range mode)
	End       int64 // unix ms (for range mode)
	Action    string // "start" or "stop" for live mode
	Cancelled bool
}

var durationRegex = regexp.MustCompile(`^(?:(\d+)h)?(?:(\d+)m)?$`)

// ParseDuration parses a duration string like "2h30m" into milliseconds.
func ParseDuration(s string) (int64, error) {
	s = strings.TrimSpace(strings.ToLower(s))
	matches := durationRegex.FindStringSubmatch(s)
	if matches == nil || (matches[1] == "" && matches[2] == "") {
		return 0, fmt.Errorf("invalid duration format: %q (expected e.g. '2h30m', '1h', '45m')", s)
	}

	var hours, minutes int64
	if matches[1] != "" {
		hours, _ = strconv.ParseInt(matches[1], 10, 64)
	}
	if matches[2] != "" {
		minutes, _ = strconv.ParseInt(matches[2], 10, 64)
	}

	return (hours*3600 + minutes*60) * 1000, nil
}

// ParseTimeRange parses a time range like "10:30am to 12:00pm" or "10:30am to now".
// Uses the provided reference time for "now" and assumes the same date.
func ParseTimeRange(s string, ref time.Time) (time.Time, time.Time, error) {
	s = strings.TrimSpace(strings.ToLower(s))
	parts := strings.SplitN(s, " to ", 2)
	if len(parts) != 2 {
		return time.Time{}, time.Time{}, fmt.Errorf("invalid range format: %q (expected 'HH:MMam to HH:MMpm' or 'HH:MMam to now')", s)
	}

	startStr := strings.TrimSpace(parts[0])
	endStr := strings.TrimSpace(parts[1])

	start, err := parseTimeOfDay(startStr, ref)
	if err != nil {
		return time.Time{}, time.Time{}, fmt.Errorf("invalid start time: %w", err)
	}

	var end time.Time
	if endStr == "now" {
		end = ref
	} else {
		end, err = parseTimeOfDay(endStr, ref)
		if err != nil {
			return time.Time{}, time.Time{}, fmt.Errorf("invalid end time: %w", err)
		}
	}

	return start, end, nil
}

func parseTimeOfDay(s string, ref time.Time) (time.Time, error) {
	formats := []string{"3:04pm", "3:04PM", "3pm", "3PM", "15:04"}
	for _, f := range formats {
		t, err := time.Parse(f, s)
		if err == nil {
			return time.Date(ref.Year(), ref.Month(), ref.Day(),
				t.Hour(), t.Minute(), 0, 0, ref.Location()), nil
		}
	}
	return time.Time{}, fmt.Errorf("cannot parse time %q", s)
}

// TimerInput is the time entry sub-menu component.
type TimerInput struct {
	mode         TimerMode
	input        string
	timerRunning bool
}

// NewTimerInput creates a new time entry sub-menu.
func NewTimerInput(timerRunning bool) TimerInput {
	return TimerInput{
		mode:         TimerModeMenu,
		timerRunning: timerRunning,
	}
}

func (t TimerInput) Init() tea.Cmd { return nil }

func (t TimerInput) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch t.mode {
		case TimerModeMenu:
			switch msg.String() {
			case "1":
				action := "start"
				if t.timerRunning {
					action = "stop"
				}
				return t, func() tea.Msg {
					return TimerResult{Mode: TimerModeLive, Action: action}
				}
			case "2":
				t.mode = TimerModeDuration
				t.input = ""
				return t, nil
			case "3":
				t.mode = TimerModeRange
				t.input = ""
				return t, nil
			case "esc":
				return t, func() tea.Msg {
					return TimerResult{Cancelled: true}
				}
			}
		case TimerModeDuration:
			switch msg.Type {
			case tea.KeyEnter:
				ms, err := ParseDuration(t.input)
				if err != nil {
					// Stay in mode, clear input
					t.input = ""
					return t, nil
				}
				return t, func() tea.Msg {
					return TimerResult{Mode: TimerModeDuration, Duration: ms}
				}
			case tea.KeyEsc:
				t.mode = TimerModeMenu
				return t, nil
			case tea.KeyBackspace:
				if len(t.input) > 0 {
					t.input = t.input[:len(t.input)-1]
				}
			default:
				if msg.Type == tea.KeyRunes {
					t.input += string(msg.Runes)
				}
			}
		case TimerModeRange:
			switch msg.Type {
			case tea.KeyEnter:
				now := time.Now()
				start, end, err := ParseTimeRange(t.input, now)
				if err != nil {
					t.input = ""
					return t, nil
				}
				return t, func() tea.Msg {
					return TimerResult{
						Mode:  TimerModeRange,
						Start: start.UnixMilli(),
						End:   end.UnixMilli(),
					}
				}
			case tea.KeyEsc:
				t.mode = TimerModeMenu
				return t, nil
			case tea.KeyBackspace:
				if len(t.input) > 0 {
					t.input = t.input[:len(t.input)-1]
				}
			default:
				if msg.Type == tea.KeyRunes {
					t.input += string(msg.Runes)
				}
			}
		}
	}
	return t, nil
}

	titleStyle := lipgloss.NewStyle().Foreground(ColorBlue).Bold(true)

			"Manual duration",
			lipgloss.NewStyle().Foreground(ColorBlue).Render("3"),
			"Time range",
			DimStyle.Render("esc cancel"),
		)
	case TimerModeDuration:
		return fmt.Sprintf("%s\n\n%s\n> %s\n\n%s",
			titleStyle.Render("Enter Duration"),
			DimStyle.Render("e.g. 2h30m, 1h, 45m"),

			t.input,
			DimStyle.Render("enter confirm · esc back"),
		)
	}
	return ""
```
```go
}

func loadData(client *api.Client, teamID, spaceID string) tea.Cmd {
	return func() tea.Msg {
		state := AppState{
			Client:  client,
			TeamID:  teamID,
			SpaceID: spaceID,

		// Get current user
		user, err := client.GetCurrentUser()
		if err != nil {
			return DataLoadedMsg{Err: fmt.Errorf("get user: %w", err)}

			return DataLoadedMsg{Err: fmt.Errorf("get lists: %w", err)}
		}

		// Get tasks from all lists
		var allTasks []api.Task
		for _, list := range lists {
			tasks, err := client.GetTasks(list.ID)
			if err != nil {
				return DataLoadedMsg{Err: fmt.Errorf("get tasks for list %s: %w", list.ID, err)}
			}
			allTasks = append(allTasks, tasks...)
		}

		state.LeafTasks = client.ResolveLeafTasks(allTasks, 5)

				return a, loadData(a.state.Client, a.state.TeamID, a.state.SpaceID)

		}
	case a.view == ViewKanban:

			task := a.kanban.SelectedTask()

				detail := NewDetail(*task, a.state)
				a.detail = &detail
				cmd = detail.Init() // Load comments
```

```go
	columns := make([]KanbanColumn, len(statuses))
	for i, s := range statuses {
		columns[i] = KanbanColumn{Status: s}
	}

	// Sort tasks into columns by status
	statusIndex := make(map[string]int)
	for i, s := range statuses {
		statusIndex[strings.ToLower(s.Status)] = i
	}

	for _, task := range tasks {
		idx, ok := statusIndex[strings.ToLower(task.Status.Status)]
		if !ok {
			// If status doesn't match, put in first column
			idx = 0
		}
		columns[idx].Tasks = append(columns[idx].Tasks, task)
	}

	return columns
}

func (k *Kanban) SetSize(w, h int) {
	k.width = w
	k.height = h
}

func (k Kanban) ShouldOpenDetail() bool { return k.openDetail }

func (k *Kanban) ClearOpenDetail() { k.openDetail = false }

func (k Kanban) SelectedTask() *api.Task {
	if k.activeCol >= len(k.columns) {
		return nil
	}
	col := k.columns[k.activeCol]
	if k.activeCard >= len(col.Tasks) {
		return nil
	}
	task := col.Tasks[k.activeCard]
	return &task
}

func (k Kanban) HandleUpdate(msg tea.Msg) (Kanban, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		if k.moving {
			return k.handleMoveMode(msg)
		}

		switch msg.String() {
		case "h", "left":
			if k.activeCol > 0 {
				k.activeCol--
				k.clampCard()
				k.ensureVisible()
			}
		case "l", "right":
			if k.activeCol < len(k.columns)-1 {
				k.activeCol++
				k.clampCard()
				k.ensureVisible()
			}
		case "j", "down":
			col := k.columns[k.activeCol]
			if k.activeCard < len(col.Tasks)-1 {
				k.activeCard++
			}
		case "k", "up":
			if k.activeCard > 0 {
				k.activeCard--
			}
		case "enter":
			k.openDetail = true
		case "m":
			if k.SelectedTask() != nil {
				k.moving = true
			}
		}
	}
	return k, nil
}

func (k Kanban) handleMoveMode(msg tea.KeyMsg) (Kanban, tea.Cmd) {
	switch msg.String() {
	case "esc":
		k.moving = false
		return k, nil
	case "h", "left":
		if k.activeCol > 0 {
			return k.moveCard(k.activeCol - 1)
		}
	case "l", "right":
		if k.activeCol < len(k.columns)-1 {
			return k.moveCard(k.activeCol + 1)
		}
	}
	// Number keys for direct column jump
	for i := 0; i < len(k.columns) && i < 9; i++ {
		if msg.String() == fmt.Sprintf("%d", i+1) {
			if i != k.activeCol {
				return k.moveCard(i)
			}
		}
	}
	return k, nil
}

func (k Kanban) moveCard(targetCol int) (Kanban, tea.Cmd) {
	task := k.SelectedTask()
	if task == nil {
		k.moving = false
		return k, nil
	}

	newStatus := k.columns[targetCol].Status.Status
	taskID := task.ID

	// Move locally
	col := &k.columns[k.activeCol]
	col.Tasks = append(col.Tasks[:k.activeCard], col.Tasks[k.activeCard+1:]...)

	task.Status = api.TaskStatus{
		Status: newStatus,
		Color:  k.columns[targetCol].Status.Color,
	}
	k.columns[targetCol].Tasks = append(k.columns[targetCol].Tasks, *task)

	k.activeCol = targetCol
	k.activeCard = len(k.columns[targetCol].Tasks) - 1
	k.moving = false
	k.ensureVisible()

	// Update via API
	return k, func() tea.Msg {
		err := k.state.Client.UpdateTask(taskID, &api.UpdateTaskRequest{Status: &newStatus})
		if err != nil {
			return StatusMsg{Text: fmt.Sprintf("Failed to move task: %v", err), IsErr: true}
		}
		return StatusMsg{Text: fmt.Sprintf("Moved to %s", newStatus)}
	}
}

func (k *Kanban) clampCard() {
	col := k.columns[k.activeCol]
	if k.activeCard >= len(col.Tasks) {
		k.activeCard = len(col.Tasks) - 1
	}
	if k.activeCard < 0 {
		k.activeCard = 0
	}
}

func (k *Kanban) ensureVisible() {
	colWidth := k.columnWidth()
	if colWidth <= 0 {
		return
	}
	previewWidth := k.width * 35 / 100
	boardWidth := k.width - previewWidth
	visibleCols := boardWidth / colWidth
	if visibleCols < 1 {
		visibleCols = 1
	}
	if k.activeCol < k.scrollX {
		k.scrollX = k.activeCol
	}
	if k.activeCol >= k.scrollX+visibleCols {
		k.scrollX = k.activeCol - visibleCols + 1
	}
}

func (k Kanban) columnWidth() int {
	previewWidth := k.width * 35 / 100
	boardWidth := k.width - previewWidth - 2
	numCols := len(k.columns)
	if numCols == 0 {

	}
	w := boardWidth / numCols
	if w < 20 {
		w = 20
	}
	if w > 40 {
		w = 40
	}
	return w
}
```

```go
func filterMyTasks(tasks []api.Task, userID int) []api.Task {
	var mine []api.Task
	for _, t := range tasks {

				break
			}
		}
	}
	return mine

func sortByPriority(tasks []api.Task) {
	sort.SliceStable(tasks, func(i, j int) bool {
		pi := priorityRank(tasks[i])
		pj := priorityRank(tasks[j])
		return pi < pj
	})
}

	if t.Priority == nil {
		return 5

	switch t.Priority.Priority {
	case "urgent":
		return 1
	case "high":
		return 2
	case "normal":
		return 3
	case "low":
		return 4
	default:
		return 5
	}
```

```go
	"fmt"

// DetailFocus tracks which panel has focus.
type DetailFocus int

const (
	FocusMain DetailFocus = iota
	FocusComments
)

// DetailOverlay tracks what overlay is shown.
type DetailOverlay int

const (
	OverlayNone DetailOverlay = iota
	OverlayStatusPicker
	OverlayPriorityPicker
	OverlayTypePicker
	OverlayAssigneePicker
	OverlayTimerInput
	OverlayTitleEditor
	OverlayDescEditor
	OverlayCommentEditor
	OverlayEditComment

	Err      error
}

// NewDetail creates a detail view for a task.

		state: state,
		task:  task,
	}

func (d Detail) Init() tea.Cmd {
	return d.loadComments()
}

func (d Detail) loadComments() tea.Cmd {
	taskID := d.task.ID
	client := d.state.Client
	return func() tea.Msg {
		comments, err := client.GetComments(taskID)
		if err != nil {
			return CommentsLoadedMsg{Err: err}
		}
		return CommentsLoadedMsg{Comments: comments}
	}
}

func (d Detail) ShouldClose() bool { return d.shouldClose }

func (d *Detail) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	// Handle overlay results first
	switch msg := msg.(type) {
	case CommentsLoadedMsg:
		if msg.Err == nil {
			d.comments = msg.Comments
		}
		return d, nil

	case ui.PickerResult:
		if msg.Cancelled {
			d.overlay = OverlayNone
			d.picker = nil
			return d, nil
		}
		return d.handlePickerResult(msg)

	case ui.EditorResult:
		if msg.Cancelled {
			d.overlay = OverlayNone
			d.editor = nil
			return d, nil
		}
		return d.handleEditorResult(msg)

	case ui.ExternalEditorResult:
		d.overlay = OverlayNone
		if msg.Err != nil {
			return d, func() tea.Msg {
				return StatusMsg{Text: msg.Err.Error(), IsErr: true}
			}
		}
		return d.updateDescription(msg.Content)

	case ui.TimerResult:
		d.overlay = OverlayNone
		d.timerInput = nil
		if msg.Cancelled {
			return d, nil
		}
		return d.handleTimerResult(msg)

	case StatusMsg:
		// Let parent handle
		return d, nil

	case TaskUpdatedMsg:
		if msg.Err != nil {
			return d, func() tea.Msg {
				return StatusMsg{Text: msg.Err.Error(), IsErr: true}
			}
		}
		return d, nil

	case tea.WindowSizeMsg:
		d.width = msg.Width
		d.height = msg.Height
		return d, nil
	}

	// Delegate to overlay if active
	if d.overlay != OverlayNone {
		return d.updateOverlay(msg)
	}

	// Handle keybinds
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "esc":
			d.shouldClose = true
			return d, nil
		case "tab":
			if d.focus == FocusMain {
				d.focus = FocusComments
			} else {
				d.focus = FocusMain
			}
			return d, nil

		// Field editing keybinds (main focus only)
		case "i":
			if d.focus == FocusMain {
				e := ui.NewEditor("Edit Title", d.task.Name)
				d.editor = &e
				d.overlay = OverlayTitleEditor
			}
		case "e":
			if d.focus == FocusMain {
				e := ui.NewEditor("Edit Description", d.task.Description)
				d.editor = &e
				d.overlay = OverlayDescEditor
			} else if d.focus == FocusComments {
				// Edit own comment
				if d.commentIdx < len(d.comments) {
					comment := d.comments[d.commentIdx]
					if comment.User.ID == d.state.CurrentUser.ID {
						e := ui.NewEditor("Edit Comment", comment.CommentText)
						d.editor = &e
						d.overlay = OverlayEditComment
					}
				}
			}
		case "E":
			if d.focus == FocusMain {
				d.overlay = OverlayDescEditor
				return d, ui.OpenExternalEditor(d.task.Description)
			}
		case "s":
			if d.focus == FocusMain {
				d.openStatusPicker()
			}
		case "p":
			if d.focus == FocusMain {
				d.openPriorityPicker()
			}
		case "y":
			if d.focus == FocusMain {
				d.openTypePicker()
			}
		case "a":
			if d.focus == FocusMain {
				d.openAssigneePicker()
			}
		case "t":
			if d.focus == FocusMain {
				ti := ui.NewTimerInput(false) // Timer running state tracked in future iteration
				d.timerInput = &ti
				d.overlay = OverlayTimerInput
			}
		case "c":
			if d.focus == FocusMain || d.focus == FocusComments {
				e := ui.NewEditor("New Comment", "")
				d.editor = &e
				d.overlay = OverlayCommentEditor
			}
		case "j", "down":
			if d.focus == FocusComments && d.commentIdx < len(d.comments)-1 {
				d.commentIdx++
			} else if d.focus == FocusMain {
				d.descScroll++
			}
		case "k", "up":
			if d.focus == FocusComments && d.commentIdx > 0 {
				d.commentIdx--
			} else if d.focus == FocusMain && d.descScroll > 0 {
				d.descScroll--
			}
		}
	}

	return d, nil
}

func (d *Detail) handlePickerResult(result ui.PickerResult) (tea.Model, tea.Cmd) {
	overlay := d.overlay
	d.overlay = OverlayNone
	d.picker = nil

	if len(result.Selected) == 0 {
		return d, nil
	}

	client := d.state.Client
	taskID := d.task.ID

	switch overlay {
	case OverlayStatusPicker:
		newStatus := result.Selected[0].ID
		d.task.Status.Status = newStatus
		return d, func() tea.Msg {
			err := client.UpdateTask(taskID, &api.UpdateTaskRequest{Status: &newStatus})
			return TaskUpdatedMsg{TaskID: taskID, Err: err}
		}

	case OverlayPriorityPicker:
		var priInt int
		fmt.Sscanf(result.Selected[0].ID, "%d", &priInt)
		d.task.Priority = &api.TaskPriority{
			ID:       result.Selected[0].ID,
			Priority: result.Selected[0].Label,
		}
		return d, func() tea.Msg {
			err := client.UpdateTask(taskID, &api.UpdateTaskRequest{Priority: &priInt})
			return TaskUpdatedMsg{TaskID: taskID, Err: err}
		}

	case OverlayTypePicker:
		d.task.TaskType = &api.TaskType{
			ID:   result.Selected[0].ID,
			Name: result.Selected[0].Label,
		}
		// ClickUp custom_item_id is set via custom_item field on task update
		var typeID int
		fmt.Sscanf(result.Selected[0].ID, "%d", &typeID)
		return d, func() tea.Msg {
			// ClickUp API uses custom_item field — send raw request
			body := map[string]interface{}{"custom_item_id": typeID}
			var resp interface{}
			err := client.Put(fmt.Sprintf("/task/%s", taskID), body, &resp)
			if err != nil {
				return StatusMsg{Text: err.Error(), IsErr: true}
			}
			return TaskUpdatedMsg{TaskID: taskID}
		}

	case OverlayAssigneePicker:
		currentIDs := make(map[int]bool)
		for _, a := range d.task.Assignees {
			currentIDs[a.ID] = true
		}
		newIDs := make(map[int]bool)
		for _, item := range result.Selected {
			var id int
			fmt.Sscanf(item.ID, "%d", &id)
			newIDs[id] = true
		}

		var add, remove []int
		for id := range newIDs {
			if !currentIDs[id] {
				add = append(add, id)
			}
		}
		for id := range currentIDs {
			if !newIDs[id] {
				remove = append(remove, id)
			}
		}

		var newAssignees []api.Assignee
		for _, item := range result.Selected {
			var id int
			fmt.Sscanf(item.ID, "%d", &id)
			newAssignees = append(newAssignees, api.Assignee{
				ID:       id,
				Username: item.Label,
			})
		}
		d.task.Assignees = newAssignees

		return d, func() tea.Msg {
			err := client.UpdateTask(taskID, &api.UpdateTaskRequest{
				Assignees: &api.AssigneeUpdate{Add: add, Remove: remove},
			})
			return TaskUpdatedMsg{TaskID: taskID, Err: err}
		}
	}

	return d, nil
}

func (d *Detail) handleEditorResult(result ui.EditorResult) (tea.Model, tea.Cmd) {
	overlay := d.overlay
	d.overlay = OverlayNone
	d.editor = nil

	switch overlay {
	case OverlayTitleEditor:
		return d.updateTitle(result.Value)
	case OverlayDescEditor:
		return d.updateDescription(result.Value)
	case OverlayCommentEditor:
		return d.createComment(result.Value)
	case OverlayEditComment:
		if d.commentIdx < len(d.comments) {
			return d.editComment(d.comments[d.commentIdx].ID, result.Value)
		}
	}
	return d, nil
}

func (d *Detail) handleTimerResult(result ui.TimerResult) (tea.Model, tea.Cmd) {
	client := d.state.Client
	taskID := d.task.ID
	teamID := d.state.TeamID

	switch result.Mode {
	case ui.TimerModeLive:
		return d, func() tea.Msg {
			var err error
			if result.Action == "start" {
				err = client.StartTimer(teamID, taskID)
			} else {
				err = client.StopTimer(teamID)
			}
			if err != nil {
				return StatusMsg{Text: err.Error(), IsErr: true}
			}
			return StatusMsg{Text: fmt.Sprintf("Timer %sed", result.Action)}
		}
	case ui.TimerModeDuration:
		return d, func() tea.Msg {
			err := client.CreateTimeEntry(taskID, &api.CreateTimeEntryRequest{
				Duration: result.Duration,
			})
			if err != nil {
				return StatusMsg{Text: err.Error(), IsErr: true}
			}
			return StatusMsg{Text: fmt.Sprintf("Added %s", ui.FormatDuration(result.Duration))}
		}
	case ui.TimerModeRange:
		return d, func() tea.Msg {
			err := client.CreateTimeEntry(taskID, &api.CreateTimeEntryRequest{
				Start:    result.Start,
				End:      result.End,
				Duration: result.End - result.Start,
			})
			if err != nil {
				return StatusMsg{Text: err.Error(), IsErr: true}
			}
			return StatusMsg{Text: "Time range recorded"}
		}
	}
	return d, nil
}

func (d Detail) View() string {
	// If overlay is active, show it centered
	if d.overlay != OverlayNone {
		var overlayView string
		switch {
		case d.picker != nil:
			overlayView = d.picker.View()
		case d.editor != nil:
			overlayView = d.editor.View()
		case d.timerInput != nil:
			overlayView = d.timerInput.View()
		}
		return lipgloss.Place(d.width, d.height,
			lipgloss.Center, lipgloss.Center,
			ui.BorderStyle.Padding(1, 2).Render(overlayView))
	}

	mainWidth := d.width * 65 / 100
	commentWidth := d.width - mainWidth

	// Main panel
	mainContent := d.renderMainPanel(mainWidth)

	// Comments panel
	commentContent := d.renderCommentPanel(commentWidth)

	mainStyle := ui.BorderStyle.Width(mainWidth - 2).Height(d.height - 4)
	commentStyle := ui.BorderStyle.Width(commentWidth - 2).Height(d.height - 4)

	if d.focus == FocusMain {
		mainStyle = ui.ActiveBorderStyle.Width(mainWidth - 2).Height(d.height - 4)
	} else {
		commentStyle = ui.ActiveBorderStyle.Width(commentWidth - 2).Height(d.height - 4)
	}

	return lipgloss.JoinHorizontal(lipgloss.Top,
		mainStyle.Render(mainContent),
		commentStyle.Render(commentContent),
	)
}

func (d Detail) renderMainPanel(width int) string {
	// Title
	title := ui.TitleStyle.Render(d.task.Name)

	// Badges
	var badges []string
	if d.task.Priority != nil {
		badges = append(badges, lipgloss.NewStyle().
			Foreground(ui.ColorRed).Render(d.task.Priority.Priority))
	}
	badges = append(badges, lipgloss.NewStyle().
		Foreground(ui.ColorYellow).Render(d.task.Status.Status))
	if d.task.TaskType != nil {
		badges = append(badges, lipgloss.NewStyle().
			Foreground(ui.ColorPurple).Render(d.task.TaskType.Name))
	}
	badgeLine := strings.Join(badges, " · ")

	// Meta
	var names []string
	for _, a := range d.task.Assignees {
		names = append(names, "@"+a.Username)
	}
	metaParts := []string{}
	if len(names) > 0 {
		metaParts = append(metaParts, strings.Join(names, ", "))
	}
	if d.task.TimeSpent > 0 {
		metaParts = append(metaParts, ui.FormatDuration(d.task.TimeSpent))
	}
	meta := ui.DimStyle.Render(strings.Join(metaParts, " · "))

	// Description with scroll offset
	desc := d.task.Description
	if desc == "" {
		desc = ui.DimStyle.Render("(no description)")
	} else {
		descLines := strings.Split(desc, "\n")
		if d.descScroll > 0 && d.descScroll < len(descLines) {
			descLines = descLines[d.descScroll:]
		}
		desc = strings.Join(descLines, "\n")
	}

	return fmt.Sprintf("%s\n%s\n%s\n\n%s\n%s",
		title, badgeLine, meta,
		ui.DimStyle.Render("DESCRIPTION"),
		desc,
	)
}

func (d Detail) renderCommentPanel(width int) string {
	header := ui.DimStyle.Render(fmt.Sprintf("COMMENTS (%d)", len(d.comments)))

	if len(d.comments) == 0 {
		return header + "\n\n" + ui.DimStyle.Render("No comments yet") +
			"\n\n" + ui.DimStyle.Render("Press c to add comment")
	}

	var commentViews []string
	for i, c := range d.comments {
		isOwn := c.User.ID == d.state.CurrentUser.ID
		isFocused := d.focus == FocusComments && i == d.commentIdx

		nameStyle := lipgloss.NewStyle().Foreground(ui.ColorBlue)
		if isOwn {
			nameStyle = lipgloss.NewStyle().Foreground(ui.ColorGreen)
		}

		entry := nameStyle.Render("@"+c.User.Username) + " " +
			ui.DimStyle.Render(c.Date) + "\n" +
			lipgloss.NewStyle().Foreground(ui.ColorFg).Render(c.CommentText)

		if isFocused {
			entry = lipgloss.NewStyle().
				Background(ui.ColorCardBg).
				Padding(0, 1).
				Render(entry)
			if isOwn {
				entry += "\n" + ui.DimStyle.Render("  e to edit")
			}
		}

		commentViews = append(commentViews, entry)
	}

	return header + "\n\n" + strings.Join(commentViews, "\n\n") +
		"\n\n" + ui.DimStyle.Render("Press c to add comment")
}
```
